void TestPopFront()
{
  std::cout << "TestPopFront..." << std::endl;

  int size = 5;
  CS170::List list;
  for (int i = 1; i <= size; i++)
    list.push_front(i);

  std::cout << list;
  while (!list.empty())
  {
    int item = list.front();
    std::cout << "First item was: " << item << std::endl;
    list.pop_front();
    std::cout << "New list:";
    std::cout << list;
    std::cout << "Items in the list: " << list.size() << std::endl;
  }
  std::cout << std::endl;
}
