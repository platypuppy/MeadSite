std::string& StringUtils::UpperCase(std::string& string)
{
  std::string::iterator it; // To walk the string 

  for (it = string.begin(); it != string.end(); ++it)
    if (*it >= 'a' && *it <= 'z')
      *it = static_cast<char>(*it - SPACE);

  return string;
}
