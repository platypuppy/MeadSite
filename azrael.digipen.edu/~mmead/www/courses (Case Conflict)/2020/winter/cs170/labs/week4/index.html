{
  namespace ListLab
  {
    struct Node
    {
      int number; // data portion
      Node *next; // pointer portion
    };

      // Adds a node to the end of the list (8)
    void AddToEnd(Node **pList, int value);

      // Adds a node to the front of the list (3)

      // Inserts a new node at the specified position (12)

  } // namespace ListLab
} // namespace CS170

#endif // LIST_H

  1. Implement the functions in a file called List.cpp. You are given 3 functions that are already implemented. Notice the helper function, MakeNode, that is in an unnamed namespace. 

    #include <iostream> // cout, endl
#include "List.h"

namespace CS170
{
  namespace ListLab
  {
    namespace 
    {
      Node* MakeNode(int value)
      {
        Node *pNode = new Node; // Allocate memory for the node
        pNode->number = value;  // Set data field (number)
        pNode->next = 0;        // Set next (no next yet)

        return pNode;
      }
    }

    void PrintList(Node *list)
    {
      while (list)
      {
        std::cout << list->number << "   ";
        list = list->next;
      }
      std::cout << std::endl;
    }

    /*!******************************************************************
      Get the number of nodes in the list.

      \param list
        The list to count.

      \return
        The number of nodes in the list.
    ********************************************************************/
    int Count(Node *list)
    {
      int count = 0; // The total number of nodes

      while (list)
      {
        count++;
        list = list->next;
      }

      return count;
    }

  } // namespace ListLab
} // namespace CS170
